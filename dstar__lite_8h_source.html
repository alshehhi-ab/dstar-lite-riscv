46struct Key{
47 int32_t k1;
48 int32_t k2;
49
50
51 // Comparison operators
52 bool operator < (const Key& other) const{
53 // check equality of k1. If both k1 values are equal check the k2 values.
54 if (k1 != other.k1) { //k1 's are different
55 return (k1 < other.k1);
56 }
57 else { // k1 's are the same, so check k2
58 return (k2 < other.k2);
59 }
60 }
61
62 bool operator > (const Key& other) const{ // opposite of less than operator
63 return (other < *this);
64 }
65
66 bool operator == (const Key& other) const{ // keys are equal when both elements are the same
67 return ((k1 == other.k1) && (k2 == other.k2));
68 }
69
70 bool operator != (const Key& other) const{ // Two kets are not equal
71 return !(*this == other); // inverse of the equality operator.
72 }
73
74};
